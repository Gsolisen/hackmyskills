    2. Create src/hms/validation.py with these components:

       ```python
       from __future__ import annotations
       from dataclasses import dataclass, field
       from pathlib import Path

       @dataclass
       class ContentError:
           file: str
           question_id: str
           message: str

       @dataclass
       class DuplicatePair:
           id_a: str
           id_b: str
           file_a: str
           file_b: str
           similarity: float  # 0.0-1.0
           reason: str  # "exact_id" or "token_overlap"

       @dataclass
       class ValidationResult:
           errors: list[ContentError] = field(default_factory=list)
           duplicates: list[DuplicatePair] = field(default_factory=list)
           files_checked: int = 0
           questions_checked: int = 0

           @property
           def ok(self) -> bool:
               return len(self.errors) == 0 and len(self.duplicates) == 0
       ```

       Token normalization function `_tokenize(text: str) -> set[str]`:
       - Lowercase the text
       - Split on whitespace and punctuation (re.split(r'[\s\W]+', text))
       - Filter tokens with len >= 2 (drop single chars)
       - Return as a set

       Token overlap function `_token_similarity(text_a: str, text_b: str) -> float`:
       - Tokenize both texts
       - If either set is empty, return 0.0
       - Compute Jaccard similarity: len(intersection) / len(union)
       - Return the float

       `_get_question_text(q: dict) -> str`: Concatenate all text fields for a question:
       - For flashcard: front + " " + back
       - For command-fill: prompt + " " + command
       - For scenario: situation + " " + question + " " + answer
       - For explain-concept: concept + " " + model_answer
       - Fallback: join all string values

       `find_duplicates(questions: list[tuple[str, dict]]) -> list[DuplicatePair]`:
       - questions is a list of (filename, question_dict) tuples
       - First pass: detect exact ID duplicates (group by q["id"])
       - Second pass: for all unique-ID pairs, compute token similarity on question text
       - Flag pairs with similarity > 0.70
       - Return list of DuplicatePair

       `validate_content_dir(content_dir: Path | None = None) -> ValidationResult`:
       - If content_dir is None, use hms.config.HMS_HOME / "content"
       - Glob for *.yaml files in content_dir
       - For each file, load with yaml.safe_load, iterate questions
       - Call loader.validate_question() on each; catch ValueError as ContentError
       - Collect all (filename, question_dict) tuples for duplicate check
       - Call find_duplicates() on the collected questions
       - Return ValidationResult with all errors, duplicates, file/question counts

       ```python
       @app.command("validate-content")
       def validate_content() -> None:
           """Validate all content YAML files: schema checks and duplicate detection."""
           from hms.validation import validate_content_dir

           result = validate_content_dir()

           if result.errors:
               console.print(f"\n[bold red]Schema Errors ({len(result.errors)}):[/bold red]")
               for err in result.errors:
                   console.print(f"  [red]\u2717[/red] {err.file} :: {err.question_id} -- {err.message}")

           if result.duplicates:
               console.print(f"\n[bold yellow]Duplicates ({len(result.duplicates)}):[/bold yellow]")
               for dup in result.duplicates:
                   if dup.reason == "exact_id":
                       console.print(f"  [yellow]\u26a0[/yellow] ID '{dup.id_a}' appears in {dup.file_a} and {dup.file_b}")
                   else:
                       console.print(
                           f"  [yellow]\u26a0[/yellow] {dup.id_a} \u2194 {dup.id_b}: {dup.similarity:.0%} overlap"
                           f" ({dup.file_a}, {dup.file_b})"
                       )

           console.print(
               f"\n[dim]Checked {result.questions_checked} questions in {result.files_checked} files.[/dim]"
           )

           if not result.ok:
               raise typer.Exit(1)
           console.print("[green]\u2713 All content valid.[/green]")
       ```
